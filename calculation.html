  2. 配列の中を全て足し算する

    // 配列arrayの値を加算する
function sum(array) {
  // 合計を格納する変数
  var sum = 0;
  // 配列の長さまで
  for (var i = 0; i < array.length; i++) {
    // ひとつずつ足していく
    sum += array[i];
  }
  // 結果を返却
  return sum;
}
const array = [1, 2, 3, 4, 5];
console.log(sum(array));
// 15

  8. 小数点以下のみを取得

    function getAfterPoint(num) {
  var arr = String(num).split(".");

  // 小数点以下がある場合と、ない場合
  if (arr[1]) {
    // 先頭に '0.' をつける
    return Number("0." + arr[1]);
  } else {
    return 0; // 小数点以下がない場合は0を返す
  }
}
